```typescript
import type { EntityFile, IndexConfig, ExportControlTiers, EntityCrossRef } from './types.js';
import { EntityType } from './types.js';

/** Metric-to-factor mapping */
const METRIC_TO_FACTOR: Record<string, string> = {
  'gpu-price-hr': 'gpu_supply',
  'primary-energy-consumption': 'energy_capacity',
  'electricity-generation': 'energy_capacity',
  'electricity-production': 'energy_capacity',
  'electricity-per-capita': 'energy_capacity',
  'renewables-share-elec': 'energy_capacity',
  'ai-capex-ttm': 'ai_capex',
};

/**
 * Percentile rank: percentage of values <= given value.
 * Returns 50 for single entity (neutral).
 */
export function percentileRank(value: number, allValues: number[]): number {
  if (allValues.length === 0) return 50;
  const count = allValues.filter(v => v <= value).length;
  return (count / allValues.length) * 100;
}

/**
 * Extract per-factor raw values from entity records.
 * For cloud-region entities with gpu-price-hr: invert (lower price = higher supply).
 * For multi-metric entities: use most recent record per metric.
 */
export function computeFactors(
  entity: EntityFile,
  crossRef: EntityCrossRef,
): Record<string, number> {
  const factors: Record<string, number> = {};

  // Scan series for most recent record per metric
  const latestByMetric = new Map<string, number>();
  for (const record of entity.series) {
    const factorKey = METRIC_TO_FACTOR[record.metric];
    if (!factorKey) continue;
    const existing = latestByMetric.get(factorKey);
    // Keep most recent (series is sorted by timestamp per CR-02 fix)
    latestByMetric.set(factorKey, record.value);
  }

  for (const [factor, value] of latestByMetric) {
    if (factor === 'gpu_supply') {
      // Invert price: lower price = higher supply score
      // Use 1/value normalized later via percentile
      factors[factor] = value > 0 ? 1 / value : 0;
    } else {
      factors[factor] = value;
    }
  }

  // Cloud region density: count cloud regions for this country
  if (entity.type === EntityType.COUNTRY) {
    const regionCount = Object.values(crossRef.cloudRegions)
      .filter(r => r.country === entity.id).length;
    factors['cloud_region_density'] = regionCount;
  }

  return factors;
}

/**
 * Normalize factors to 0-100 via percentile ranking within same entity type.
 * Returns Map<entityId, Record<factorName, normalizedScore>>.
 */
export function normalizeFactors(
  entityFactors: Map<string, Record<string, number>>,
  entityTypes: Map<string, EntityType>,
  targetType: EntityType,
): Map<string, Record<string, number>> {
  const result = new Map<string, Record<string, number>>();

  // Filter to entities of the target type
  const typeEntities = Array.from(entityFactors.entries())
    .filter(([id]) => entityTypes.get(id) === targetType);

  if (typeEntities.length === 0) return result;

  // Collect all factor names present
  const allFactorNames = new Set<string>();
  for (const [, factors] of typeEntities) {
    for (const key of Object.keys(factors)) {
      allFactorNames.add(key);
    }
  }

  // For each factor, compute percentile ranks
  for (const factorName of allFactorNames) {
    const values = typeEntities
      .map(([id, factors]) => ({ id, value: factors[factorName] }))
      .filter(v => v.value !== undefined);

    const allVals = values.map(v => v.value);

    for (const { id, value } of values) {
      if (!result.has(id)) result.set(id, {});
      result.get(id)![factorName] = percentileRank(value, allVals);
    }
  }

  return result;
}

/**
 * Compute composite score per D-01:
 * Score = Sum(weight_i * normalized_factor_i) * risk_multiplier
 *
 * Re-weights proportionally when factors are missing (per D-03).
 */
export function computeCompositeScore(
  normalized: Record<string, number>,
  weights: Record<string, number>,
  riskMultiplier: number,
): { score: number; presentWeight: number } {
  let weightedSum = 0;
  let presentWeight = 0;

  for (const [factor, weight] of Object.entries(weights)) {
    if (normalized[factor] !== undefined) {
      weightedSum += weight * normalized[factor];
      presentWeight += weight;
    }
  }

  // Re-weight proportionally (per D-03)
  const rawScore = presentWeight > 0 ? weightedSum / presentWeight : 0;

  return {
    score: rawScore * riskMultiplier,
    presentWeight,
  };
}

/**
 * Compute entity confidence score per D-07, D-08, D-09.
 */
export function computeConfidence(
  entity: EntityFile,
  factorNames: string[],
  config: IndexConfig,
): { confidence: number; dataCompleteness: 'full' | 'partial' } {
  // Base confidence: weighted average of record confidences
  const totalWeight = entity.series.length;
  const baseConfidence = totalWeight > 0
    ? entity.series.reduce((sum, r) => sum + r.confidence, 0) / totalWeight
    : 0;

  // Missing factor penalty (D-08)
  const expectedFactors = 5; // gpu_supply, energy, density, capex, risk
  const missingCount = Math.max(0, expectedFactors - factorNames.length);
  const missingPenalty = missingCount * config.confidence.missingFactorPenalty;

  // Staleness penalty (D-09)
  const now = Date.now();
  let stalenessPenalty = 0;
  for (const record of entity.series) {
    const ageMs = now - new Date(record.timestamp).getTime();
    const ageDays = ageMs / (1000 * 60 * 60 * 24);
    if (ageDays > config.confidence.veryStaleDays) {
      stalenessPenalty = Math.max(stalenessPenalty, config.confidence.veryStalePenalty);
    } else if (ageDays > config.confidence.staleDays) {
      stalenessPenalty = Math.max(stalenessPenalty, config.confidence.stalePenalty);
    }
  }

  const confidence = Math.max(
    config.confidence.minConfidence,
    Math.round(baseConfidence - missingPenalty - stalenessPenalty),
  );

  const dataCompleteness = factorNames.length < 3 ? 'partial' : 'full';

  return { confidence, dataCompleteness };
}

/**
 * Aggregate cloud-region data to parent country.
 * Returns new entity factors for countries.
 */
export function aggregateToCountries(
  entities: Map<string, EntityFile>,
  crossRef: EntityCrossRef,
): Map<string, Record<string, number>> {
  const countryFactors = new Map<string, Record<string, number>>();

  // Collect cloud-region factors by country
  const regionByCountry = new Map<string, Record<string, number>[]>();

  for (const [id, entity] of entities) {
    if (entity.type !== EntityType.CLOUD_REGION) continue;
    const regionInfo = crossRef.cloudRegions[id];
    if (!regionInfo) continue;

    const factors = computeFactors(entity, crossRef);
    if (Object.keys(factors).length === 0) continue;

    const existing = regionByCountry.get(regionInfo.country) ?? [];
    existing.push(factors);
    regionByCountry.set(regionInfo.country, existing);
  }

  // Average GPU_Supply across regions per country
  for (const [countryId, regions] of regionByCountry) {
    const gpuValues = regions
      .map(r => r.gpu_supply)
      .filter(v => v !== undefined);

    if (gpuValues.length > 0) {
      const avg = gpuValues.reduce((s, v) => s + v, 0) / gpuValues.length;
      if (!countryFactors.has(countryId)) countryFactors.set(countryId, {});
      countryFactors.get(countryId)!.gpu_supply = avg;
    }
  }

  return countryFactors;
}

/**
 * Aggregate cloud-region data to parent city.
 * Returns new entity factors for cities derived from crossref city field.
 * Cities are virtual entities — no separate entity files needed.
 */
export function aggregateToCities(
  entities: Map<string, EntityFile>,
  crossRef: EntityCrossRef,
): Map<string, Record<string, number>> {
  const cityFactors = new Map<string, Record<string, number>>();

  // Collect cloud-region factors by city
  const regionByCity = new Map<string, Record<string, number>[]>();

  for (const [id, entity] of entities) {
    if (entity.type !== EntityType.CLOUD_REGION) continue;
    const regionInfo = crossRef.cloudRegions[id];
    if (!regionInfo?.city) continue;

    const factors = computeFactors(entity, crossRef);
    if (Object.keys(factors).length === 0) continue;

    const existing = regionByCity.get(regionInfo.city) ?? [];
    existing.push(factors);
    regionByCity.set(regionInfo.city, existing);
  }

  // Average GPU_Supply and count density per city
  for (const [cityId, regions] of regionByCity) {
    const gpuValues = regions
      .map(r => r.gpu_supply)
      .filter(v => v !== undefined);

    const cityEntry: Record<string, number> = {};

    if (gpuValues.length > 0) {
      cityEntry.gpu_supply = gpuValues.reduce((s, v) => s + v, 0) / gpuValues.length;
    }
    cityEntry.cloud_region_density = regions.length;

    cityFactors.set(cityId, cityEntry);
  }

  return cityFactors;
}
```

Key design decisions:
- **Pure functions:** No I/O, no file reads. Compiler passes data in, gets results out.
- **Metric-to-factor mapping:** METRIC_TO_FACTOR translates scraper metric names to index factor names. For GPU supply, price is inverted (1/value) so lower price = higher supply score.
- **Percentile ranking per entity type:** Countries ranked against countries, cloud regions against cloud regions.
- **Graceful degradation (D-03):** computeCompositeScore re-weights by dividing by presentWeight.
- **Staleness penalty:** Uses max penalty across all records (not per-record penalty).

Actually, re-reading the current pipeline code more carefully: the scraper loop already collects records. The issue is that records are written per-scraper in the loop. To support multi-source merging, restructure the pipeline:

```typescript
async function run(): Promise<void> {
  console.log('[pipeline] Starting...');
  const meta = await loadMeta();

  // 1. Discover and run all scrapers
  await discoverScrapers();
  const scrapers = getScrapers();

  // 2. Collect all records across all scrapers
  const allRecords = new Map<string, NormalizedRecord[]>();
  let totalRecords = 0;

  for (const scraper of scrapers) {
    console.log(`[pipeline] Running scraper: ${scraper.name}`);
    try {
      const records = await scraper.fetch();
      totalRecords += records.length;

      for (const [entityId, entityRecords] of groupByEntity(records)) {
        const existing = allRecords.get(entityId) ?? [];
        existing.push(...entityRecords);
        allRecords.set(entityId, existing);
      }
    } catch (err) {
      console.error(`[pipeline] Scraper ${scraper.name} failed: ${err}`);
    }
  }

  // 3. Write merged entity files
  let written = 0;
  let skipped = 0;

  for (const [entityId, records] of allRecords) {
    if (records.length === 0) continue;
    const entityType = records[0].entity.type;
    const changed = await writeEntityIfChanged(entityId, entityType, records, meta);
    if (changed) written++;
    else skipped++;
  }

  // 4. Compile (with guard per CR-01 fix)
  if (written === 0 && Object.keys(meta.entities).length === 0) {
    console.warn('[pipeline] No data produced and no existing entities. Skipping compile.');
    return;
  }

  await compile(DATA_DIR);

  // 5. Write metadata
  meta.lastRun = new Date().toISOString();
  await fs.writeFile(META_PATH, JSON.stringify(meta, null, 2));

  console.log(`[pipeline] Done. Records: ${totalRecords}, Written: ${written}, Skipped: ${skipped}`);
}
```

Key changes:
- Collect all scraper outputs first, then merge by entity ID
- Each scraper failure is caught individually (one failing scraper doesn't crash pipeline)
- Entity type determined from first record (all records for same entity have same type)
- Written/skipped counts reflect merged entities, not per-scraper entities
