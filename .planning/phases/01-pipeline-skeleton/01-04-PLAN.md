```typescript
import fs from 'node:fs/promises';
import path from 'node:path';
import { EntityType } from './types.js';
import type { EntityFile, CompiledEntity, NormalizedRecord } from './types.js';

/**
 * Compile all entity files into aggregate outputs.
 * @param dataDir Base data directory (default: 'public/data'). Entity files
 *   are read from {dataDir}/entities/**/*.json. Outputs are written to {dataDir}/.
 */
export async function compile(dataDir = 'public/data'): Promise<void> {
  const entities = await loadAllEntities(dataDir);
  await writeLatest(entities, dataDir);
  await writeRankings(entities, dataDir);
  await writeHistory(entities, dataDir);
}
```

- Use `fs.readdir` recursively or `glob` to find all `.json` files under `{dataDir}/entities/`
- Parse each file as JSON and validate it has `id`, `type`, `latest` fields

writeLatest implementation:
- Build `{ generated: new Date().toISOString(), entities: { [id]: CompiledEntity } }`

writeRankings implementation:

writeHistory implementation:

Create `scripts/run-pipeline.ts` as the single entry point for the data pipeline. This is what `pnpm run pipeline` executes.

The pipeline orchestrator:
1. Loads pipeline metadata from `public/data/_pipeline-meta.json` (or initializes empty meta)
2. Discovers and runs all registered scrapers
3. Groups each scraper's records by entity ID
4. For each entity group: computes SHA-256 hash, compares with stored hash, writes entity file only if changed
5. Compiles aggregate outputs (latest.json, rankings.json, history.json)
6. Updates and writes pipeline metadata
7. Logs summary (written/skipped counts)

```typescript
import { discoverScrapers, getScrapers } from './registry.js';
import { hashRecords } from './hash.js';
import { compile } from './compiler.js';
import type { NormalizedRecord, PipelineMeta, EntityFile } from './types.js';
import { EntityType } from './types.js';
import fs from 'node:fs/promises';
import path from 'node:path';

const DATA_DIR = 'public/data';
const META_PATH = path.join(DATA_DIR, '_pipeline-meta.json');

async function loadMeta(): Promise<PipelineMeta> {
  try {
    const raw = await fs.readFile(META_PATH, 'utf-8');
    return JSON.parse(raw);
  } catch {
    return { lastRun: '', entities: {} };
  }
}

function groupByEntity(records: NormalizedRecord[]): Map<string, NormalizedRecord[]> {
  const groups = new Map<string, NormalizedRecord[]>();
  for (const record of records) {
    const existing = groups.get(record.entity.id) ?? [];
    existing.push(record);
    groups.set(record.entity.id, existing);
  }
  return groups;
}

async function writeEntityIfChanged(
  entityId: string,
  entityType: EntityType,
  records: NormalizedRecord[],
  meta: PipelineMeta,
): Promise<boolean> {
  const newHash = hashRecords(records);
  const existingHash = meta.entities[entityId]?.hash;

  if (newHash === existingHash) {
    console.log(`[skip] ${entityId} unchanged (hash match)`);
    return false;
  }

  const filePath = path.join(DATA_DIR, 'entities', entityType, `${entityId}.json`);
  await fs.mkdir(path.dirname(filePath), { recursive: true });

  const entityFile: EntityFile = {
    id: entityId,
    type: entityType,
    latest: records[records.length - 1],
    series: records,
    _hash: newHash,
    _updatedAt: new Date().toISOString(),
  };

  await fs.writeFile(filePath, JSON.stringify(entityFile, null, 2));
  meta.entities[entityId] = { hash: newHash, updatedAt: new Date().toISOString() };
  return true;
}

async function run(): Promise<void> {
  console.log('[pipeline] Starting...');
  const meta = await loadMeta();

  // 1. Discover and run scrapers
  await discoverScrapers();
  const scrapers = getScrapers();
  let written = 0, skipped = 0;

  for (const scraper of scrapers) {
    console.log(`[pipeline] Running scraper: ${scraper.name}`);
    const records = await scraper.fetch();

    // 2. Group records by entity
    const byEntity = groupByEntity(records);

    // 3. Write entity files (with hash check)
    for (const [entityId, entityRecords] of byEntity) {
      // Use the entity type from the first record
      const entityType = entityRecords[0].entity.type;
      const changed = await writeEntityIfChanged(entityId, entityType, entityRecords, meta);
      if (changed) written++; else skipped++;
    }
  }

  // 4. Compile aggregate outputs
  await compile(DATA_DIR);

  // 5. Write pipeline metadata
  meta.lastRun = new Date().toISOString();
  await fs.writeFile(META_PATH, JSON.stringify(meta, null, 2));

  console.log(`[pipeline] Done. Written: ${written}, Skipped: ${skipped}`);
}

run().catch(err => {
  console.error('[pipeline] Fatal error:', err);
  process.exit(1);
});
```

Key implementation details:
1. The entity type comes from the NormalizedRecord's entity.type field (already set by the scraper)
2. hashRecords is called on the records array BEFORE writing -- per Pitfall 5, hash the records not the entity file
3. The entity file includes `_hash` and `_updatedAt` which are NOT included in the hash computation
4. All file paths are relative to project root (where `pnpm run pipeline` executes)
5. Error handling: top-level catch-all at entry point with process.exit(1) per RESEARCH.md Pattern 6
6. Console logging: `[pipeline]` prefix for orchestration, `[skip]` prefix for hash-match skips

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-01 | Tampering | Entity file integrity in git | accept | Entity files committed to git provide tamper evidence via commit history. Pipeline metadata tracks hashes for change detection. Acceptable for public pricing data. |
| T-04-02 | Denial of Service | Compiler loading large entity set | accept | Phase 1 has ~36 AWS regions (max ~50 entity files). File sizes are small (1-5KB each). Not a concern until Phase 2 with 100+ entities. |
| T-04-03 | Information Disclosure | _pipeline-meta.json exposes entity hashes | accept | SHA-256 hashes are one-way; they reveal nothing about entity contents. Metadata is operational data, not sensitive. |
| T-04-04 | Tampering | Malformed entity file crashes compiler | mitigate | loadAllEntities validates each file has required fields (id, type, latest) before adding to map. Malformed files are logged and skipped, not crashing the pipeline. Per ASVS V5. |
</threat_model>
